## 2. 架构设计

### 2.1 架构分层

```
┌─────────────────────────────────────────────┐
│              View Layer (视图层)            │
│  Pages / Components / Layouts              │
├─────────────────────────────────────────────┤
│              Business Layer (业务层)        │
│  Composables / Services                    │
├─────────────────────────────────────────────┤
│              Data Layer (数据层)            │
│  Store (Pinia) / API                       │
├─────────────────────────────────────────────┤
│              Infrastructure (基础设施)      │
│  Router / Axios / Utils                    │
└─────────────────────────────────────────────┘
```

### 2.2 核心设计原则

- **组件化**: 将UI拆分为独立、可复用的组件
- **状态管理**: 使用Pinia管理全局状态（用户信息、权限）
- **路由守卫**: 控制页面访问权限
- **API封装**: 统一封装HTTP请求，处理响应和错误
- **响应式设计**: 支持桌面端和移动端

## 6. 状态管理方案

### 6.1 User Store

```javascript
// stores/user.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  // 状态
  const token = ref('')
  const userId = ref(null)
  const realName = ref('')
  const role = ref('VISITOR')
  
  // 计算属性
  const isLoggedIn = computed(() => !!token.value)
  const isRecruiter = computed(() => role.value === 'RECRUITER')
  const isApprover = computed(() => role.value === 'APPROVER')
  const isVisitor = computed(() => role.value === 'VISITOR')
  
  // 方法
  function login(data) {
    token.value = data.token
    userId.value = data.userId
    realName.value = data.realName
    role.value = data.role
    localStorage.setItem('user', JSON.stringify(data))
  }
  
  function logout() {
    token.value = ''
    userId.value = null
    realName.value = ''
    role.value = 'VISITOR'
    localStorage.removeItem('user')
  }
  
  function initFromStorage() {
    const stored = localStorage.getItem('user')
    if (stored) {
      const data = JSON.parse(stored)
      login(data)
    }
  }
  
  return {
    token, userId, realName, role,
    isLoggedIn, isRecruiter, isApprover, isVisitor,
    login, logout, initFromStorage
  }
})
```

### 6.2 Position Store

```javascript
// stores/position.js
import { defineStore } from 'pinia'
import { ref } from 'vue'
import * as positionService from '../services/position'

export const usePositionStore = defineStore('position', () => {
  const list = ref([])
  const current = ref(null)
  const loading = ref(false)
  const total = ref(0)
  
  async function fetchList(params) {
    loading.value = true
    try {
      const res = await positionService.getList(params)
      list.value = res.data
      total.value = res.total || res.data.length
    } catch (err) {
      console.error('获取岗位列表失败:', err)
    } finally {
      loading.value = false
    }
  }
  
  async function fetchById(id) {
    loading.value = true
    try {
      const res = await positionService.getById(id)
      current.value = res.data
      return current.value
    } catch (err) {
      console.error('获取岗位详情失败:', err)
    } finally {
      loading.value = false
    }
  }
  
  async function add(position) {
    try {
      await positionService.add(position)
      await fetchList({})
    } catch (err) {
      throw err
    }
  }
  
  async function update(position) {
    try {
      await positionService.update(position)
      await fetchList({})
    } catch (err) {
      throw err
    }
  }
  
  async function remove(id) {
    try {
      await positionService.remove(id)
      await fetchList({})
    } catch (err) {
      throw err
    }
  }
  
  return {
    list, current, loading, total,
    fetchList, fetchById, add, update, remove
  }
})
```

## 7. API接口对接

### 7.1 Axios封装

```javascript
// utils/axios.js
import axios from 'axios'
import { useUserStore } from '../stores/user'

const instance = axios.create({
  baseURL: '/api',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json;charset=UTF-8'
  }
})

// 请求拦截器
instance.interceptors.request.use(
  config => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers['X-Authorization'] = `Bearer ${userStore.token}`
    }
    if (userStore.role) {
      config.headers['X-User-Role'] = userStore.role
    }
    return config
  },
  error => {
    return Promise.reject(error)
  }
)

// 响应拦截器
instance.interceptors.response.use(
  response => {
    return response.data
  },
  error => {
    if (error.response?.status === 401) {
      const userStore = useUserStore()
      userStore.logout()
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

export default instance
```

// services/position.js
import axios from '../utils/axios'

export async function getList(params) {
  return await axios.get('/webapi/positions', { params })
}

export async function getById(id) {
  return await axios.get(`/webapi/positions/${id}`)
}

export async function add(data) {
  return await axios.post('/webapi/positions', data)
}

export async function update(id, data) {
  return await axios.put(`/webapi/positions/${id}`, data)
}

export async function remove(id) {
  return await axios.delete(`/webapi/positions/${id}`)
}

export async function submitApproval(id) {
  return await axios.post(`/webapi/positions/${id}/submit`)
}

export async function approve(id, data) {
  return await axios.post(`/webapi/positions/${id}/approve`, data)
}

export async function close(id) {
  return await axios.post(`/webapi/positions/${id}/close`)
}

export async function upload(file) {
  const formData = new FormData()
  formData.append('file', file)
  return await axios.post('/webapi/positions/upload', formData, {
    headers: { 'Content-Type': 'multipart/form-data' }
  })
}

## 9. 安全性考虑

### 9.1 认证安全
- 使用JWT令牌进行身份验证
- Token存储在localStorage
- 请求头携带Token和角色信息

### 9.2 权限控制
- 路由守卫检查登录状态和角色
- 动态渲染菜单（根据角色显示不同菜单项）
- 按钮级权限控制（根据角色显示/隐藏操作按钮）

### 9.3 输入验证
- 前端表单验证（必填、长度、格式）
- 防止XSS攻击（对用户输入进行转义）
- 防止CSRF攻击（使用Token验证）

### 9.4 错误处理
- 统一错误处理（401跳转到登录页）
- 友好的错误提示
- 日志记录
